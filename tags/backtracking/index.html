    What Is Backtracking Backtracking Template Example class Solution { private List<String> parentheses = new ArrayList<>(); public List<String> generateParenthesis(int n) { backtrack("",n,n); System.out.println(parentheses.toString()); return parentheses; } private void backtrack(String s, int left, int right) { if (left == 0 && right == 0) { parentheses.add(s); return; } if (left > right || left < 0) { return; } String tmp = s + "("; backtrack(tmp, left-1, right); tmp = s + ")"; backtrack(tmp, left, right-1); return; } }
